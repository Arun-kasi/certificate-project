const int sensor_pin = A0; float sensor_output; float voltage; float tempC; float tempF; void setup() { Serial.begin(9600); } void loop() { sensor_output = analogRead(sensor_pin); voltage = (sensor_output / 1024.0) * 5.0; tempC = (voltage - 0.5) * 100.0; tempF = (tempC * 9.0 / 5.0) + 32.0; Serial.print("Temperature in °C: "); Serial.println(tempC); Serial.print("Temperature in °F: "); Serial.println(tempF); Serial.println("------------------------"); delay(3000); }
